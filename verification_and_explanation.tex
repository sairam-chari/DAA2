\section{Verification Result (Current Code State)}
\subsection{Invariant Checks Used}
For a polygon with holes, a valid triangulation should satisfy:
\begin{enumerate}
  \item Triangle interiors are inside walkable space.
  \item No triangle edge intersects blocked boundary edges except at shared endpoints.
  \item Area conservation: total triangle area equals walkable area.
  \item Triangle count formula (after hole-bridging to a simple polygon) should be consistent with vertex count of the bridged simple polygon.
\end{enumerate}

\subsection{Executed Counterexample}
Tested case:
\begin{itemize}
  \item Outer square: $(0,0),(20,0),(20,20),(0,20)$
  \item One hole: $(7,7),(13,7),(13,13),(7,13)$
\end{itemize}
Observed from program output and direct area audit:
\begin{itemize}
  \item Reported triangles: $4$
  \item Walkable doubled-area: $728$
  \item Sum of triangle doubled-areas: $560$
\end{itemize}
Hence area conservation fails.

\textbf{Conclusion:} the current implementation is \textbf{not fully correct} yet for general polygons with holes.

\section{Theorems and Mathematical Facts Used}
\begin{theorem}[Jordan Curve Theorem, algorithmic use]
A simple closed polygonal chain partitions the plane into interior and exterior regions.
\end{theorem}
\textit{Use in code:} basis for ray-casting in \texttt{point\_in\_ring} and region membership in \texttt{point\_in\_region}.

\begin{lemma}[Orientation and Segment Intersection Test]
Given points $a,b,c,d$, orientation predicates and on-segment checks determine whether segments $ab$ and $cd$ intersect.
\end{lemma}
\textit{Use in code:} \texttt{orientation}, \texttt{on\_segment}, \texttt{segments\_intersect}.

\begin{theorem}[Ear Existence Theorem]
Every simple polygon with at least 4 vertices has at least two ears.
\end{theorem}
\textit{Use in code:} justifies iterative ear clipping in \texttt{ear\_clip\_triangulation}.

\begin{theorem}[Art Gallery Upper Bound with Holes]
For a polygon with $n$ total vertices and $h$ holes, an upper bound on point guards is
\[
\left\lfloor \frac{n + 2h}{3} \right\rfloor.
\]
\end{theorem}
\textit{Use in code:} computed in \texttt{solve} as output metric \texttt{k}.

\begin{lemma}[3-Coloring of Triangulation Graph]
A maximal outerplanar triangulation graph is 3-colorable; choosing the smallest color class gives an upper bound guard set for simple polygons.
\end{lemma}
\textit{Use in code:} heuristic guard selection via \texttt{color\_outerplanar\_graph} and \texttt{choose\_guard\_positions}.

\section{Complexity Notes}
\begin{itemize}
  \item Primitive tests are constant time each.
  \item Sweep-line helper section is designed around sorted status operations, but implemented with list scans/sorts in places; practical complexity can exceed strict $O(n \log n)$.
  \item Ear clipping is typically $O(n^2)$ in worst case.
\end{itemize}
